########################################################################################
#############               生成 base64_live555 库文件                            ############## 
########################################################################################
LIST (APPEND base64_live555_LIB_SRCS
    ${CMAKE_CURRENT_SOURCE_DIR}/Base64.cpp
    )
    
   
       
########################################################################################
#############               设置 base64_live555 库版本号                          ############## 
########################################################################################
set(base64_live555_LIB_VERSION     "1.0.0")
set(base64_live555_LIB_SOVERSION   "1")



########################################################################################
#############              添加头文件目录                                 ############## 
########################################################################################
include_directories(  
    ${PROJECT_SOURCE_DIR}/includes  
    ${PROJECT_SOURCE_DIR}/Crypto
    ${PROJECT_SOURCE_DIR}/Crypto/Base64_live555   
)  



########################################################################################
#############               添加 链接库文件目录                           ############## 
########################################################################################
link_directories(  
    /usr/local/lib   
    )  


    
########################################################################################
#############           生成目标库文件 并 标明版本号                      ############## 
########################################################################################
IF (MODULE_base64_live555)

    #生成动态库  静态库则 STATIC  
    add_library (base64_live555lib  SHARED          ${base64_live555_LIB_SRCS})  
    add_library (base64_live555s    STATIC          ${base64_live555_LIB_SRCS})  
    
    # 设置版本号，SOVERSION为API版本号
    set_target_properties(base64_live555lib     PROPERTIES 
                                        VERSION     ${base64_live555_LIB_VERSION} 
                                        SOVERSION   ${base64_live555_LIB_SOVERSION} )
ELSE (MODULE_base64_live555)
    MESSAGE(STATUS "Not Include base64_live555 module.")
ENDIF (MODULE_base64_live555)



########################################################################################
#############                 生成目标库测试文件                          ############## 
########################################################################################
IF (TEST_MODULE_FLAG AND MODULE_base64_live555)
    add_executable(TestBase64_live555.elf    base64_live555Test.cpp)
    add_dependencies(TestBase64_live555.elf          base64_live555lib loglib)
    target_link_libraries(TestBase64_live555.elf     base64_live555lib loglib)

ELSE (TEST_MODULE_FLAG AND MODULE_base64_live555)
    MESSAGE(STATUS "Not Include base64_live555 module.")
ENDIF (TEST_MODULE_FLAG AND MODULE_base64_live555)
